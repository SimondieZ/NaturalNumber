
	---- RESET TABLE SEQUENCES AND DELETE DATA FROM TABLES-----
	
DELETE FROM github.numbers;
DELETE FROM github.users;

ALTER SEQUENCE github.numbers_id_seq RESTART WITH 1;	
UPDATE github.numbers SET id=nextval('github.numbers_id_seq');

ALTER SEQUENCE github.users_id_seq RESTART WITH 1;	
UPDATE github.users SET id=nextval('github.users_id_seq');

CREATE SCHEMA github;
CREATE TABLE github.numbers (
    id bigserial NOT NULL,
    value bigint NOT NULL,
    binary_notation character varying,
    roma_notation character varying,
    description character varying,
    divisors integer[],
    PRIMARY KEY (id)
);

CREATE TABLE github.users
(
    id bigserial NOT NULL,
    email character varying(255) NOT NULL,
    first_name character varying(50) NOT NULL,
    last_name character varying(100),
    password character varying(255) NOT NULL,
    role character varying(20) NOT NULL DEFAULT 'USER',
    status character varying(20) DEFAULT 'ACTIVE',
    PRIMARY KEY (id),
    UNIQUE (email)
);
